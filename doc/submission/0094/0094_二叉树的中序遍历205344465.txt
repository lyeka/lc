var output []int
func inorderTraversal(root *TreeNode) []int {
	output = []int{}
	fn(root)
	return output
}

func fn(root *TreeNode) {
	if root == nil {
		return
	}
	fn(root.Left)
	output = append(output, root.Val)
	fn(root.Right)
}
